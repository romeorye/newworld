LCheckBoxGroup은 form 영역에 구현할 경우에는 단순한 출력용으로 사용할 필요는 없다. HTML input 태그로 내용을 만들어서 사용한다. 샘플 경로 : /rui/sample/general/ui/form/checkboxgroupSample.html
sample: 기본 샘플이다.
var checkBoxGroup = new Rui.ui.form.LCheckBoxGroup({ applyTo: 'checkBoxGroup', name: 'checkBoxGroup', items: [ { label: 'Apple', name: 'fruits', value: 'apple' }, { label: 'Grape', value: 'Grape' }, { label: 'Strawberry', value: 'Strawberry' } ] });
