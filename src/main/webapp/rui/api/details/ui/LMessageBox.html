window의 alert/confirm/prompt 메시지에 html 태그를 적용하여 출력할 수 있는 기능 샘플 경로 : /rui/sample/general/ui/container/messageboxSample.html
sample: alert 기본 샘플
Rui.alert('DevOn RichUI 입니다.');
sample: alert에 handler를 탑재하는 샘플
Rui.alert({ text: 'DevOn RichUI 입니다.', handler: function() { Rui.get('input1').focus(); } });
sample: confirm 기본 샘플
Rui.confirm({ text: '실행하시겠습니까?', buttons: [ { text: '실행합니다.' }, { text: '취소' } ], handlerYes: function() { Rui.get('input1').focus(); }, handlerNo: Rui.emptyFn });
sample: prompt 기본 샘플
Rui.prompt({ text: '입력하세요.', handler: function(value) { Rui.get('input1').focus(); } });
